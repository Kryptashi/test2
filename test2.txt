<div id="faqContainer"></div>

<script>
// Sample FAQ dataset
const faqs = [
    { question: "How do I reset my password?", answer: "Go to settings and click 'Reset Password'.", topic: "account", product: "webex", lang: "en" },
    { question: "Como redefinir minha senha?", answer: "Vá para configurações e clique em 'Redefinir senha'.", topic: "account", product: "webex", lang: "pt" },
    { question: "How to troubleshoot call quality?", answer: "Check your network connection and restart the app.", topic: "call", product: "webex", lang: "en" },
    { question: "What is Webex Contact Center?", answer: "It is a cloud-based contact center solution.", topic: "general", product: "webex", lang: "en" }
];

// Read query parameters
const params = new URLSearchParams(window.location.search);
const topicParam = params.get('topic');
const productParam = params.get('product');
const langParam = params.get('lang');

// Filter FAQs based on query parameters
let filteredFaqs = faqs.filter(faq => {
    return (!topicParam || faq.topic === topicParam) &&
           (!productParam || faq.product === productParam) &&
           (!langParam || faq.lang === langParam);
});

// Render FAQs
const faqContainer = document.getElementById('faqContainer');
function renderFaqs(list) {
    faqContainer.innerHTML = '';
    if (list.length === 0) {
        faqContainer.innerHTML = '<p>No FAQs found for your criteria.</p>';
        return;
    }
    list.forEach(faq => {
        const div = document.createElement('div');
        div.className = 'faq-item';
        div.innerHTML = `<div class="faq-question">${faq.question}</div>
                         <div class="faq-answer">${faq.answer}</div>`;
        faqContainer.appendChild(div);
    });
}
renderFaqs(filteredFaqs);

// Search functionality
document.getElementById('searchInput').addEventListener('input', function() {
    const query = this.value.toLowerCase();
    const searchedFaqs = filteredFaqs.filter(faq =>
        faq.question.toLowerCase().includes(query) ||
        faq.answer.toLowerCase().includes(query)
    );
    renderFaqs(searchedFaqs);
});
</script>
